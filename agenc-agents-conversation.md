 #### 3. Go Implementation Skeleton (The Gateway Service)                                                                                                                                           
                                                                                                                                                                                                    
 ```go                                                                                                                                                                                              
   // internal/ingestion/handler.go                                                                                                                                                                 
   package ingestion                                                                                                                                                                                
                                                                                                                                                                                                    
   import (                                                                                                                                                                                         
      "context"                                                                                                                                                                                     
      "net/http"                                                                                                                                                                                    
      "time"                                                                                                                                                                                        
                                                                                                                                                                                                    
      "github.com/gin-gonic/gin"                                                                                                                                                                    
      "go.uber.org/zap"                                                                                                                                                                             
      "google.golang.org/protobuf/proto"                                                                                                                                                            
                                                                                                                                                                                                    
      pb "github.com/smarttrigger/backend/proto/ingestion/v1"                                                                                                                                       
      "github.com/smarttrigger/backend/internal/vault"                                                                                                                                              
      "github.com/smarttrigger/backend/internal/queue"                                                                                                                                              
   )                                                                                                                                                                                                
                                                                                                                                                                                                    
   type Handler struct {                                                                                                                                                                            
      logger    *zap.Logger                                                                                                                                                                         
      vaultClient *vault.Client                                                                                                                                                                     
      publisher  *queue.Publisher // Publishes to Kafka/NATS for Classifier                                                                                                                         
   }                                                                                                                                                                                                
                                                                                                                                                                                                    
   func NewHandler(logger *zap.Logger, vaultClient *vault.Client, publisher *queue.Publisher) *Handler {                                                                                            
      return &Handler{logger: logger, vaultClient: vaultClient, publisher: publisher}                                                                                                               
   }                                                                                                                                                                                                
                                                                                                                                                                                                    
   // SubmitUrgencyEvent handles POST /events/urgency                                                                                                                                               
   // Content-Type MUST be application/protobuf                                                                                                                                                     
   func (h *Handler) SubmitUrgencyEvent(c *gin.Context) {                                                                                                                                           
      ctx := c.Request.Context()                                                                                                                                                                    
                                                                                                                                                                                                    
      // 1. Enforce Protobuf Content-Type                                                                                                                                                           
      if c.ContentType() != "application/protobuf" {                                                                                                                                                
         h.logger.Warn("Invalid Content-Type", zap.String("got", c.ContentType()))                                                                                                                  
         c.AbortWithStatusJSON(http.StatusUnsupportedMediaType, gin.H{"error": "Content-Type must be application/protobuf"})                                                                        
         return                                                                                                                                                                                     
      }                                                                                                                                                                                             
                                                                                                                                                                                                    
      // 2. Parse Protobuf                                                                                                                                                                          
      var req pb.IntakeEvent                                                                                                                                                                        
      body, _ := c.GetRawData()                                                                                                                                                                     
      if err := proto.Unmarshal(body, &req); err != nil {                                                                                                                                           
         h.logger.Warn("Protobuf unmarshal failed", zap.Error(err))                                                                                                                                 
         c.AbortWithStatusJSON(http.StatusBadRequest, gin.H{"error": "Invalid protobuf payload"})                                                                                                   
         return                                                                                                                                                                                     
      }                                                                                                                                                                                             
                                                                                                                                                                                                    
      // 3. Validate Provider Identity (mTLS + API Key already verified by middleware)                                                                                                              
      // 4. Validate Patient Pseudo-ID exists in Vault (Prevents orphan events)                                                                                                                     
      exists, err := h.vaultClient.ValidatePseudoID(ctx, req.PatientPseudoId)                                                                                                                       
      if err != nil || !exists {                                                                                                                                                                    
         h.logger.Warn("Invalid or unknown patient_pseudo_id", zap.String("pid", req.PatientPseudoId))                                                                                              
         c.AbortWithStatusJSON(http.StatusBadRequest, gin.H{"error": "Invalid patient identifier"})                                                                                                 
         return                                                                                                                                                                                     
      }                                                                                                                                                                                             
                                                                                                                                                                                                    
      // 5. Validate Clinical Payload (Business Logic Constraints)                                                                                                                                  
      if err := validateClinicalPayload(&req); err != nil {                                                                                                                                         
         c.AbortWithStatusJSON(http.StatusBadRequest, gin.H{"error": err.Error()})                                                                                                                  
         return                                                                                                                                                                                     
      }                                                                                                                                                                                             
                                                                                                                                                                                                    
      // 6. Publish to Async Queue (Kafka/NATS) for Classification Engine                                                                                                                           
      // We do NOT call the Classifier synchronously. This gateway must return < 100ms.                                                                                                             
      trackingID := uuid.New()                                                                                                                                                                      
      job := &queue.ClassificationJob{                                                                                                                                                              
         TrackingID:   trackingID,                                                                                                                                                                  
         IntakeEvent:  &req,                                                                                                                                                                        
         ReceivedAt:   time.Now().UnixMilli(),                                                                                                                                                      
      }                                                                                                                                                                                             
                                                                                                                                                                                                    
      if err := h.publisher.Publish(ctx, "urgency.classify", job); err != nil {                                                                                                                     
         h.logger.Error("Failed to queue classification job", zap.Error(err))                                                                                                                       
         c.AbortWithStatusJSON(http.StatusInternalServerError, gin.H{"error": "Classification queue unavailable"})                                                                                  
         return                                                                                                                                                                                     
      }                                                                                                                                                                                             
                                                                                                                                                                                                    
      // 7. Return 202 Accepted Immediately                                                                                                                                                         
      resp := &pb.AcceptanceReceipt{                                                                                                                                                                
         TrackingId:    trackingID.String(),                                                                                                                                                        
         Status:        "QUEUED",                                                                                                                                                                   
         ReceivedAtUnix: time.Now().Unix(),                                                                                                                                                         
      }                                                                                                                                                                                             
      respBytes, _ := proto.Marshal(resp)                                                                                                                                                           
      c.Data(http.StatusAccepted, "application/protobuf", respBytes)                                                                                                                                
   }                                                                                                                                                                                                
                                                                                                                                                                                                    
   func validateClinicalPayload(req *pb.IntakeEvent) error {                                                                                                                                        
      p := req.ClinicalPayload                                                                                                                                                                      
      if p == nil { return errors.New("clinical_payload required") }                                                                                                                                
      if len(p.CptCode) != 5 { return errors.New("cpt_code must be 5 digits") }                                                                                                                     
      // ... add regex validation for ICD10, vital ranges ...                                                                                                                                       
      return nil                                                                                                                                                                                    
   }
 ```                                                                                                                                                                                                
